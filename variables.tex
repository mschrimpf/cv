% helper
% from http://tex.stackexchange.com/a/44938/39530
\makeatletter
\newcommand{\DoIfNotEmpty}[1]{%
  \begingroup
  \sbox0{#1}%
  \ifdim\wd0=\z@
    \endgroup
    \expandafter\@gobble
  \else
    \endgroup
    \expandafter\@firstofone
  \fi}
\makeatother
